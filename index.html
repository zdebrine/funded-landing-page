Get started with just a few lines of code

Register an app in the Dashboard to get a client ID and client secret, and then add the SDK to an existing app via Cocoapods or download our starter project.

import CoreLocation
import SpotSense

let spotsense = SpotSense(clientID: "client-id", clientSecret: "client-secret")

class ViewController: UIViewController, CLLocationManagerDelegate, SpotSenseDelegate {
  let locationManager : CLLocationManager = CLLocationManager()
  override func viewDidLoad() {
    super.viewDidLoad()

    // get location permissions
    locationManager.requestAlwaysAuthorization()
    locationManager.delegate = self
    spotsense.delegate = self

    spotsense.startMonitoring() // this is where the magic happens
  }
}
